SET FEEDBACK OFF

TTITLE '회원 플레이리스트'
COLUMN 음원번호 HEADING '음원|번호'
COLUMN 음원명
COLUMN 장르
COLUMN 가수
COLUMN 추가일시 HEADING '추가|일시'

SELECT NUM AS 음원번호, SOUNDTRACK.NAME AS 음원명, GENRE AS 장르, UPLOAD_SINGER AS 가수, PLAYLIST_DATE AS 추가일시
FROM PLAYLIST, SOUNDTRACK, MUSICAPP_USER
WHERE MUSICAPP_USER.USER_ID = '&회원ID'
AND PLAYLIST.USER_ID = MUSICAPP_USER.USER_ID
AND PLAYLIST.LIST_NUM = SOUNDTRACK.NUM
ORDER BY &정렬기준
/

TTITLE OFF
CLEAR COLUMNS
