TTITLE '출시된 음향기기|제휴사이트 명단'
COLUMN RELEASED HEADING '출시일'
COLUMN PNAME HEADING '제품명'
COLUMN CNAME HEADING '출시회사'
COLUMN NUM HEADING '품번' FORMAT 000000000
COLUMN SNAME HEADING '제휴사이트'
COLUMN EDATE HEADING '등록|일시'
COLUMN PRICE HEADING '가격' FORMAT 999,999,999
COLUMN STOCK_NUM HEADING '재고' FORMAT 999,999
BREAK ON RELEASED ON PNAME ON CNAME

SELECT TO_CHAR(PRO_DATE,'YYYY/MM/DD') AS RELEASED, PRODUCT.NAME AS PNAME, PRODUCT.COMPANY AS CNAME, PRODUCT.NUM, SITE.NAME AS SNAME, ENROLL_DATE AS EDATE, PRICE, STOCK_NUM
FROM SITE, PRODUCT
WHERE TO_CHAR(PRODUCT.PRO_DATE,'YYYY') = '&년도'
AND SITE.NUM = PRODUCT.ENROLL_SITE
ORDER BY PRO_DATE
/

TTITLE OFF
CLEAR COLUMNS
CLEAR BREAK
