ALTER SESSION SET NLS_DATE_FORMAT = 'YYYY/MM/DD HH24:MI:SS';
SET PAGESIZE 30
SET LINESIZE 100
SET FEEDBACK OFF

TTITLE '회원 월정액 가입 내역'
COLUMN NAME HEADING '이름'
COLUMN YMON HEADING '년월'
COLUMN MONCA HEADING '월정액|구분'
COLUMN PAYMENT_TYPE HEADING '결제수단'
COLUMN FEE HEADING '결제|금액' FORMAT 999,999
BREAK ON NAME ON YMON

SELECT NAME, TO_CHAR(JOIN_DATE,'YYYY/MM') AS YMON, MONTHLY_CATEGORY AS MONCA, PAYMENT_TYPE, FEE
FROM JOIN, MUSICAPP_USER, MONTHLYFEE
WHERE TO_CHAR(JOIN_DATE,'YYYY') >= '&몇년도부터'
AND TO_CHAR(JOIN_DATE,'YYYY') <= '&몇년도까지'
AND JOIN.USER_ID = MUSICAPP_USER.USER_ID
AND MONTHLY_CATEGORY = CATEGORY
AND JOIN.USER_ID = '&회원ID'
ORDER BY YMON
/

TTITLE OFF
CLEAR COLUMNS
CLEAR BREAK
