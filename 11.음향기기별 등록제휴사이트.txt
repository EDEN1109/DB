SET FEEDBACK OFF

TTITLE '제휴사이트 명단'
COLUMN PNAME HEADING '제품명' 
COLUMN CATEGORY HEADING '카테고리'
COLUMN NUM HEADING '품번' FORMAT 000000000
COLUMN SNAME HEADING '제휴사이트'
COLUMN ENROLL_DATE HEADING '등록|일시'
COLUMN PRICE HEADING '가격' FORMAT 999,999,999
COLUMN STOCK_NUM HEADING '재고' FORMAT 999,999
BREAK ON PNAME ON CATEGORY

SELECT PRODUCT.NAME AS PNAME, CATEGORY, PRODUCT.NUM AS NUM, SITE.NAME AS SNAME, ENROLL_DATE, PRICE, STOCK_NUM
FROM SITE, PRODUCT
WHERE PRODUCT.NAME LIKE '%&제품명%'
AND SITE.NUM = PRODUCT.ENROLL_SITE
ORDER BY PNAME, PRICE
/

TTITLE OFF
CLEAR COLUMNS
CLEAR BREAK
